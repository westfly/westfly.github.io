function batch_url_normalize($filename)
{
    $handle = fopen($filename, "r") or die ("open fail");
    if ($handle) {
        while (!feof($handle)) {
            $buffer = fgets($handle, 40960);
            process_line(trim($buffer));
        }
        fclose($handle);
    }
}
